<?hh

namespace Pi\Odm;
use Pi\Odm\DataProperty;
use Pi\Odm\DataCollection;

class DataPropertyCollection {

  protected $columnFromName = Map{};

  protected $columnName = Map{};

  protected $list = Set{};

  public function __construct(protected DataCollection $collection)
  {

  }

  public function contains($name) : bool
  {
    return $this->columnFromName->contains($name);
  }

  public function add(?DataProperty $dataProperty = null) : DataProperty
  {
    if(is_null($dataProperty)) {
      $dataProperty = new DataProperty();
    }

    if(empty($dataProperty->getPropertyName())){
      $dataProperty->setPropertyName('random');
    }

    if(!is_null($dataProperty->getCollection())){
      throw new \Exception('The property already belong to this or another collection');
    }

    $dataProperty->setCollection($this->collection);
    $this->registerName($dataProperty->getPropertyName(), $dataProperty);
    $ordinal = $this->list->add($dataProperty);
    $dataProperty->setOrdinal($ordinal);

    return $dataProperty;
  }

  private function registerName(string $name, DataProperty $property)
  {
    try {
      $this->columnFromName[$name] = $property;
    }
    catch(\Exception $ex){
      throw new \Exception(
        sprintf('A DataProperty %s already belongs to this DataCollection', $name)
      );
    }

    if(!$this->columnName->contains($name)) {
      $this->columnName[$name] = Set{$name};
    } else {
      $this->columnName[$name]->add($name);
    }
  }
}
